    <script>
      $(document).ready(function() {
        // Verificar proporção ao carregar a página
        checkProportion();
        
        // Verificar proporção ao redimensionar a janela
        $(window).resize(function() {
          checkProportion();
        });
        
        function checkProportion() {
          var screenWidth = $(window).width();
          var screenHeight = $(window).height();
          
          var ratio = screenWidth / screenHeight;
          var threshold = 1.5; // proporção desejada
          
          // Verificar se a proporção foi atingida
          if (ratio >= threshold) {
            showOverlay();
          } else {
            hideOverlay();
          }
        }
        
        function showOverlay() {
          $('#overlay').show();
        }
        
        function hideOverlay() {
          $('#overlay').hide();
        }
      });
    </script>
